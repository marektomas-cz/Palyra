Usage: palyra onboarding wizard [OPTIONS]

Options:
      --path <PATH>

      --force

      --flow <FLOW>
          [possible values: quickstart, manual, remote]
      --non-interactive

      --accept-risk

      --json

      --workspace-root <WORKSPACE_ROOT>

      --auth-method <AUTH_METHOD>
          [possible values: api-key, anthropic-api-key, minimax-api-key, skip, existing-config]
      --api-key-env <API_KEY_ENV>

      --api-key-stdin
          Read one model-provider API key from stdin; requires --non-interactive for scripted wizard runs
      --api-key-prompt

      --deployment-profile <DEPLOYMENT_PROFILE>
          [possible values: local, single-vm, worker-enabled]
      --bind-profile <BIND_PROFILE>
          [possible values: loopback-only, public-tls]
      --daemon-port <DAEMON_PORT>

      --grpc-port <GRPC_PORT>

      --quic-port <QUIC_PORT>

      --tls-scaffold <TLS_SCAFFOLD>
          [possible values: none, bring-your-own, self-signed]
      --tls-cert-path <TLS_CERT_PATH>

      --tls-key-path <TLS_KEY_PATH>

      --remote-base-url <REMOTE_BASE_URL>

      --admin-token-env <ADMIN_TOKEN_ENV>

      --admin-token-stdin
          Read one admin token from stdin; requires --non-interactive for scripted wizard runs
      --admin-token-prompt

      --remote-verification <REMOTE_VERIFICATION>
          [possible values: none, server-cert, gateway-ca]
      --pinned-server-cert-sha256 <PINNED_SERVER_CERT_SHA256>

      --pinned-gateway-ca-sha256 <PINNED_GATEWAY_CA_SHA256>

      --ssh-target <SSH_TARGET>

      --skip-health

      --skip-channels

      --skip-skills

  -h, --help
          Print help

Global Options:
      --profile <PROFILE>
          Use the named CLI profile from the local profile registry
      --expect-profile <EXPECT_PROFILE>
          Require the resolved active profile name to match this value
      --config <CONFIG_PATH>
          Use this palyra.toml config file
      --state-root <STATE_ROOT>
          Use this runtime state root directory
  -v, --verbose...
          Increase logging verbosity (-v => debug, -vv => trace)
      --log-level <LOG_LEVEL>
          Set diagnostic log verbosity [default: info] [possible values: error, warn, info, debug, trace]
      --output-format <OUTPUT_FORMAT>
          Select text, JSON, or NDJSON output for automation [default: text] [possible values: text, json, ndjson]
      --plain
          Disable styled terminal output
      --no-color
          Disable ANSI color output
      --allow-profile-mismatch
          Allow commands to continue when the selected profile differs from context
      --allow-strict-profile-actions
          Permit strict profile-scoped actions after explicit profile checks
