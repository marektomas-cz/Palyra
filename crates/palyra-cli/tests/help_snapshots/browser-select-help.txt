Set the selected value of a <select> element

Usage: palyra browser select [OPTIONS] --selector <SELECTOR> --value <VALUE> <SESSION_ID>

Arguments:
  <SESSION_ID>

Options:
      --selector <SELECTOR>
      --value <VALUE>
  -h, --help                 Print help

Global Options:
      --profile <PROFILE>

      --expect-profile <EXPECT_PROFILE>

      --config <CONFIG_PATH>

      --state-root <STATE_ROOT>

  -v, --verbose...
          Increase logging verbosity (-v => debug, -vv => trace)
      --log-level <LOG_LEVEL>
          [default: info] [possible values: error, warn, info, debug, trace]
      --output-format <OUTPUT_FORMAT>
          [default: text] [possible values: text, json, ndjson]
      --plain

      --no-color

      --allow-profile-mismatch

      --allow-strict-profile-actions
