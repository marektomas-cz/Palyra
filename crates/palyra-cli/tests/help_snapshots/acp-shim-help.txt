Run the legacy NDJSON ACP compatibility shim

Usage: palyra acp shim [OPTIONS]

Options:
      --grpc-url <GRPC_URL>
      --token <TOKEN>
      --principal <PRINCIPAL>
      --device-id <DEVICE_ID>
      --channel <CHANNEL>
      --session-id <SESSION_ID>
      --session-key <SESSION_KEY>
      --session-label <SESSION_LABEL>
      --require-existing
      --reset-session
      --run-id <RUN_ID>
      --prompt <PROMPT>
      --prompt-stdin
      --allow-sensitive-tools
      --ndjson-stdin
  -h, --help                           Print help

Global Options:
      --profile <PROFILE>
      --config <CONFIG_PATH>
      --state-root <STATE_ROOT>
  -v, --verbose...                     Increase logging verbosity (-v => debug, -vv => trace)
      --log-level <LOG_LEVEL>          [default: info] [possible values: error, warn, info, debug, trace]
      --output-format <OUTPUT_FORMAT>  [default: text] [possible values: text, json, ndjson]
      --plain
      --no-color
