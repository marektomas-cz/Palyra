Usage: palyra cron add [OPTIONS] --name <NAME> --schedule-type <SCHEDULE_TYPE> --schedule <SCHEDULE> <--prompt <PROMPT>|--prompt-stdin>

Options:
      --name <NAME>

      --prompt <PROMPT>

      --prompt-stdin

      --schedule-type <SCHEDULE_TYPE>
          [possible values: cron, every, at]
      --schedule <SCHEDULE>
          Schedule payload: cron expression for cron, RFC3339 timestamp for at, or milliseconds/duration such as 300000 or 5m for every
      --enabled
          Create the scheduled job enabled immediately; omitted jobs are created disabled
      --concurrency <CONCURRENCY>
          [default: forbid] [possible values: forbid, replace, queue-one]
      --retry-max-attempts <RETRY_MAX_ATTEMPTS>
          [default: 1]
      --retry-backoff-ms <RETRY_BACKOFF_MS>
          [default: 1000]
      --misfire <MISFIRE>
          [default: skip] [possible values: skip, catch-up]
      --jitter-ms <JITTER_MS>
          [default: 0]
      --owner <OWNER>

      --channel <CHANNEL>

      --session-key <SESSION_KEY>

      --session-label <SESSION_LABEL>

      --json

  -h, --help
          Print help

Global Options:
      --profile <PROFILE>
          Use the named CLI profile from the local profile registry
      --expect-profile <EXPECT_PROFILE>
          Require the resolved active profile name to match this value
      --config <CONFIG_PATH>
          Use this palyra.toml config file
      --state-root <STATE_ROOT>
          Use this runtime state root directory
  -v, --verbose...
          Increase logging verbosity (-v => debug, -vv => trace)
      --log-level <LOG_LEVEL>
          Set diagnostic log verbosity [default: info] [possible values: error, warn, info, debug, trace]
      --output-format <OUTPUT_FORMAT>
          Select text, JSON, or NDJSON output for automation [default: text] [possible values: text, json, ndjson]
      --plain
          Disable styled terminal output
      --no-color
          Disable ANSI color output
      --allow-profile-mismatch
          Allow commands to continue when the selected profile differs from context
      --allow-strict-profile-actions
          Permit strict profile-scoped actions after explicit profile checks
